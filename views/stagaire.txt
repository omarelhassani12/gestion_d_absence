<!-- JavaScript code -->
<script>
    document.addEventListener('DOMContentLoaded', function () {
        // When the "New Stagiaire" button is clicked, show the modal
        document.getElementById('new-stagiaire-button').addEventListener('click', function () {
            const modal = new bootstrap.Modal(document.getElementById('modal-report'));
            modal.show();
        });

        // Handle form submission
        document.getElementById('create-stagiaire-form').addEventListener('submit', function (event) {
            event.preventDefault(); // Prevent the form from submitting

            // Get the form data
            const form = document.getElementById('create-stagiaire-form');
            const formData = new FormData(form);

            // Send the form data via AJAX
            fetch('/stagaire/create', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json' // Add the Content-Type header
                },
                body: JSON.stringify(Object.fromEntries(formData)) // Convert FormData to JSON
            })
                .then(response => response.json())
                .then(data => {
                    // Rest of the code remains the same
                    if (data.success) {
                        // Stagiaire created successfully
                        alert('Stagiaire created successfully');
                        window.location.href = window.location.href; // Reload the page
                    } else {
                        // Error creating stagiaire
                        console.error('An error occurred while creating the stagiaire: ' + data.error);
                    }
                })
                .catch(error => {
                    console.error(error);
                    console.error('An error occurred while creating the stagiaire');
                })
                .finally(() => {
                    const modalBackdrop = document.getElementsByClassName('modal-backdrop');
                    const modalContainer = document.getElementsByClassName('modal');

                    if (modalBackdrop.length > 0) {
                        modalBackdrop[0].parentNode.removeChild(modalBackdrop[0]); // Remove the modal backdrop from the DOM
                    }

                    if (modalContainer.length > 0) {
                        modalContainer[0].parentNode.removeChild(modalContainer[0]); // Remove the modal container from the DOM
                    }
                });
        });

        // When the modal is hidden, reset the form
        document.getElementById('modal-report').addEventListener('hidden.bs.modal', function () {
            document.getElementById('create-stagiaire-form').reset();
        });
    });

    // Handle delete button click
    function confirmDelete(event, stagiaireId) {
        event.preventDefault(); // Prevent form submission
        if (confirm("Are you sure you want to delete this record?")) {
            fetch(`/stagaire/${stagiaireId}`, {
                method: 'DELETE'
            })
                .then(response => response.json())
                .then(data => {
                    if (data.success) {
                        // Stagiaire deleted successfully
                        alert('Stagiaire deleted successfully');
                        window.location.href = window.location.href; // Reload the page
                    } else {
                        // Error deleting stagiaire
                        console.error('An error occurred while deleting the stagiaire: ' + data.error);
                    }
                })
                .catch(error => {
                    console.error(error);
                    console.error('An error occurred while deleting the stagiaire');
                });
        }
    }
</script>
